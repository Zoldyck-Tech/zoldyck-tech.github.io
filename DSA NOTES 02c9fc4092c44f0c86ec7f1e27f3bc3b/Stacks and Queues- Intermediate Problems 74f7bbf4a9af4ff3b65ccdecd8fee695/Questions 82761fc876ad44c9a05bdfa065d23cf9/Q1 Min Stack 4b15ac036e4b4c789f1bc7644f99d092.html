Solutions

Basically we need to Design and implement a stack and do operations like push,Â pop ,Â top , andÂ retrieve the minimum element in constant time.

Approach :

  • Using Linked List we can do
  • Create Linked List Node class with insertfront , pop , top
  • for min value , keep a minhead and add the values lesser than the min head value

Python :

class Node():
    def __init__(self,val,next=None):
        self.val = val
        self.next = None
        
        
class LinkedList():
    def __init__(self):
        self.head = None
        self.minhead = None
                        
    def insertfront(self,val): 
        newNode = Node(val)
        if self.head: 
            temp = self.head
            newNode.next = temp
            self.head = newNode
        else:
            self.head = newNode
        
        newminhead = Node(val)
        if self.minhead == None:
            self.minhead = newminhead
        
        elif self.minhead.val == None:
            self.minhead = newminhead
            
        elif val <= self.minhead.val:
            oldmin = self.minhead
            newminhead.next = oldmin
            self.minhead = newminhead
            
    def pop(self):
        if self.head:
            remove = self.head
            if remove.val == self.minhead.val:
                oldmin = self.minhead
                self.minhead = oldmin.next               
            
            newhead = remove.next
            self.head = newhead
        else:
            return
        
    def top(self):
        if self.head:
            return self.head.val
        else:
            return -1


class MinStack:
    # @param x, an integer
    # @return an integer
    def __init__(self):
        self.ll = LinkedList()
        
    def push(self, x):
        return self.ll.insertfront(x)
    
    # @return nothing
    def pop(self):
        return self.ll.pop()

    # @return an integer
    def top(self):
        return self.ll.top()
        

    # @return an integer
    def getMin(self):
        temp = self.ll.minhead

        if temp:
            return self.ll.minhead.val
        else:
            return -1
